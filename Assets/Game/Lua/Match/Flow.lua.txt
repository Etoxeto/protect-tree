local flow_config = require("Config.MatchFlow")
local wave_config = require("Config.Waves")

local M = {}

local phase_durations = {
    Preparation = flow_config.preparation_seconds,
    Battle = flow_config.battle_defense_seconds,
    JointDefenseIntro = flow_config.joint_defense_intro_seconds or 1.5,
    JointDefense = flow_config.joint_defense_seconds
        or flow_config.battle_defense_seconds,
    BossPreparation = flow_config.boss_preparation_seconds,
    BossBattle = flow_config.boss_defense_seconds,
    Settlement = flow_config.settlement_seconds,
}

local current_phase = nil
local current_wave = 0
local remaining_seconds = 0
local is_started = false
local is_finished = false
local result = "InProgress"

local pending_events = {}

local function require_positive_integer(value, field_name)
    if type(value) ~= "number" or value <= 0 or value % 1 ~= 0 then
        error(field_name .. " must be a positive integer: " .. tostring(value))
    end
end

local function get_wave_schedule()
    local schedule = wave_config.schedule or {}
    local normal_wave_count =
        schedule.normal_wave_count or flow_config.normal_wave_count
    local boss_wave = schedule.boss_wave or flow_config.boss_wave

    return normal_wave_count, boss_wave
end

local function has_wave_pool(wave)
    if type(wave_config.wave_pools) == "table" then
        return wave_config.wave_pools[wave] ~= nil
    end

    return wave_config[wave] ~= nil
end

local function validate_config()
    local normal_wave_count, boss_wave = get_wave_schedule()
    require_positive_integer(normal_wave_count, "normal_wave_count")
    require_positive_integer(boss_wave, "boss_wave")

    if boss_wave ~= normal_wave_count + 1 then
        error("boss_wave must immediately follow normal waves")
    end

    for wave = 1, normal_wave_count do
        if not has_wave_pool(wave) then
            error("Wave pool not found for normal wave: " .. tostring(wave))
        end
    end

    if not has_wave_pool(boss_wave) then
        error("Wave pool not found for boss wave: " .. tostring(boss_wave))
    end

    for phase, duration in pairs(phase_durations) do
        if type(duration) ~= "number" or duration <= 0 then
            error("Phase duration must be positive: " .. phase)
        end
    end
end

local function get_phase_duration(phase)
    return phase_durations[phase] or 0
end

local function emit_phase_changed()
    local event = {
        type = "PhaseChanged",
        wave = current_wave,
        phase = current_phase,
    }

    if result ~= "InProgress" then
        event.result = result
    end

    table.insert(pending_events, event)
end

local function enter_phase(phase, wave)
    current_phase = phase
    current_wave = wave or current_wave
    remaining_seconds = get_phase_duration(phase)
    emit_phase_changed()
end

local function enter_next_timed_phase()
    if current_phase == "Preparation" then
        enter_phase("Battle")
        return
    end

    if current_phase == "Settlement" then
        local normal_wave_count, boss_wave = get_wave_schedule()
        if current_wave < normal_wave_count then
            enter_phase("Preparation", current_wave + 1)
        else
            enter_phase("BossPreparation", boss_wave)
        end
        return
    end

    if current_phase == "JointDefenseIntro" then
        enter_phase("JointDefense")
        return
    end

    if current_phase == "BossPreparation" then
        enter_phase("BossBattle")
        return
    end

    error("Timed phase cannot advance: " .. tostring(current_phase))
end

function M.start()
    validate_config()
    pending_events = {}
    is_started = true
    is_finished = false
    result = "InProgress"
    enter_phase("Preparation", 1)
end

function M.update(delta_time)
    if not is_started or is_finished then
        return
    end

    if current_phase == "Battle"
        or current_phase == "JointDefense"
        or current_phase == "BossBattle" then
        remaining_seconds = math.max(0, remaining_seconds - delta_time)
        return
    end

    remaining_seconds = remaining_seconds - delta_time

    while remaining_seconds <= 0 and not is_finished
        and current_phase ~= "Battle"
        and current_phase ~= "JointDefense"
        and current_phase ~= "BossBattle" do
        local overflow = -remaining_seconds
        enter_next_timed_phase()
        remaining_seconds = remaining_seconds - overflow
    end
end

function M.is_phase_timeout()
    if not is_started or is_finished then
        return false
    end

    if current_phase ~= "Battle"
        and current_phase ~= "JointDefense"
        and current_phase ~= "BossBattle" then
        return false
    end

    return remaining_seconds <= 0
end

function M.complete_battle()
    if not is_started or is_finished or current_phase ~= "Battle" then
        return false
    end

    enter_phase("Settlement")
    return true
end

function M.begin_joint_defense()
    if not is_started or is_finished or current_phase ~= "Battle" then
        return false
    end

    enter_phase("JointDefenseIntro")
    return true
end

function M.complete_joint_defense()
    if not is_started or is_finished or current_phase ~= "JointDefense" then
        return false
    end

    enter_phase("Settlement")
    return true
end

function M.complete_preparation()
    if not is_started or is_finished then
        return false
    end

    if current_phase ~= "Preparation" and current_phase ~= "BossPreparation" then
        return false
    end

    enter_next_timed_phase()
    return true
end

function M.debug_enter_boss_preparation()
    if not is_started or is_finished or current_phase ~= "Preparation" then
        return false
    end

    local _, boss_wave = get_wave_schedule()
    enter_phase("BossPreparation", boss_wave)
    return true
end

function M.end_match(match_result)
    if not is_started or is_finished then
        return false
    end

    if match_result ~= "Victory" and match_result ~= "Defeat" then
        error("Invalid match result: " .. tostring(match_result))
    end

    result = match_result
    is_finished = true
    enter_phase("End")
    return true
end

function M.complete_boss(match_result)
    if current_phase ~= "BossBattle" then
        return false
    end

    return M.end_match(match_result)
end

function M.shutdown()
    is_started = false
end

function M.get_snapshot()
    return {
        phase = current_phase,
        wave = current_wave,
        remaining_seconds = math.max(remaining_seconds, 0),
        is_finished = is_finished,
        result = result,
    }
end

function M.drain_events()
    local events = pending_events
    pending_events = {}
    return events
end

return M
