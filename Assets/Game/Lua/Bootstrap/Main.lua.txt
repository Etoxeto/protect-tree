function M.start_local_multiplayer(player_count)
    if type(player_count) ~= "number"
        or player_count < 1
        or player_count % 1 ~= 0 then
        error("player_count must be a positive integer")
    end

    if player_count > max_players then
        error("player_count exceeds max supported players: " .. player_count)
    end

    local player_ids = {}
    for player_id = 1, player_count do
        table.insert(player_ids, player_id)
    end

    -- 该入口仅用于本地多人原型/测试：重建权威 Session，不改变 Lua 环境和生命周期实验模块。
    session.shutdown()
    session.start({
        player_ids = player_ids,
    })
end

function M.reload_module(module_name)
    if module_name ~= lifecycle_demo_module then
        error("Module is not reloadable: " .. module_name)
    end

    local previous_module = lifecycle_demo
    previous_module.shutdown()
    package.loaded[module_name] = nil

    local loaded, replacement_or_error = pcall(require, module_name)

    if not loaded then
        package.loaded[module_name] = previous_module
        previous_module.start()
        error("Failed to reload " .. module_name .. ": " .. replacement_or_error)
    end

    lifecycle_demo = replacement_or_error
    lifecycle_demo.start()
end
