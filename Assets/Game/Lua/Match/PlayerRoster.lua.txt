local player_config = require("Match.PlayerConfig")

local M = {}

local active_player_configs = {}
local players_by_id = {}
local ordered_player_ids = {}
local alive_count = 0
local pending_events = {}

local function require_non_negative_integer(value, field_name)
    if type(value) ~= "number"
        or value < 0
        or value % 1 ~= 0 then
        error(
            field_name
            .. " must be a non-negative integer: "
            .. tostring(value)
        )
    end
end

local function reset(session_options)
    active_player_configs, ordered_player_ids =
        player_config.get_active_player_configs(session_options)
    players_by_id = {}
    alive_count = 0
    pending_events = {}
end

function M.start(session_options)
    reset(session_options)

    for _, player_id in ipairs(ordered_player_ids) do
        local config = active_player_configs[player_id]
        if type(player_id) ~= "number" or player_id % 1 ~= 0 then
            error("Player ID must be an integer: " .. tostring(player_id))
        end

        if config.max_health == nil or config.max_health <= 0 then
            error("Invalid max_health for player: " .. player_id)
        end

        if config.starting_gold == nil or config.starting_gold < 0 then
            error("Invalid starting_gold for player: " .. player_id)
        end

        require_non_negative_integer(config.round_reward, "round_reward")
        require_non_negative_integer(
            config.round_reward_per_wave_increment or 0,
            "round_reward_per_wave_increment"
        )
    end

    for _, player_id in ipairs(ordered_player_ids) do
        local config = active_player_configs[player_id]

        players_by_id[player_id] = {
            player_id = player_id,
            health = config.max_health,
            max_health = config.max_health,
            gold = config.starting_gold,
            status = "Alive",
            is_ready = false,
        }

        alive_count = alive_count + 1
    end
end

function M.handle_event(event)
    if event.type ~= "PlayerLeakResolved" then
        return
    end

    local player_id = event.player_id

    if player_id == nil then
        error("PlayerLeakResolved missing player_id")
    end

    local player = players_by_id[player_id]

    if player == nil then
        error("Player not found: " .. tostring(player_id))
    end

    require_non_negative_integer(
        event.initial_leak_count,
        "initial_leak_count"
    )

    require_non_negative_integer(
        event.rescued_count,
        "rescued_count"
    )
    require_non_negative_integer(
        event.final_leak_count,
        "final_leak_count"
    )

    if event.rescued_count > event.initial_leak_count then
        error("rescued_count cannot exceed initial_leak_count")
    end

    if event.final_leak_count
        ~= event.initial_leak_count - event.rescued_count then
        error("final_leak_count does not match leak resolution")
    end

    if event.final_leak_count == 0 then
        return
    end

    if player.status ~= "Alive" then
        return
    end

    local damage = math.min(event.final_leak_count, player.health)
    player.health = player.health - damage

    table.insert(pending_events, {
        type = "PlayerDamaged",
        player_id = player.player_id,
        damage = damage,
        leak_count = event.final_leak_count,
        health = player.health,
        wave = event.wave,
    })

    if player.health == 0 then
        player.status = "Eliminated"
        alive_count = alive_count - 1

        table.insert(pending_events, {
            type = "PlayerEliminated",
            player_id = player.player_id,
            health = player.health,
            wave = event.wave,
        })
    end
end

function M.set_ready(player_id, is_ready)
    local player = players_by_id[player_id]
    if player == nil then
        error("Player not found: " .. tostring(player_id))
    end

    if player.status ~= "Alive" then
        error("Eliminated player cannot change ready state: " .. player_id)
    end

    if type(is_ready) ~= "boolean" then
        error("is_ready must be a boolean")
    end

    if player.is_ready == is_ready then
        return
    end

    player.is_ready = is_ready
    table.insert(pending_events, {
        type = "PlayerReadyChanged",
        player_id = player_id,
        is_ready = is_ready,
    })
end

function M.reset_ready()
    for _, player_id in ipairs(ordered_player_ids) do
        local player = players_by_id[player_id]
        if player.is_ready then
            player.is_ready = false
            table.insert(pending_events, {
                type = "PlayerReadyChanged",
                player_id = player_id,
                is_ready = false,
            })
        end
    end
end

function M.are_all_alive_players_ready()
    if alive_count == 0 then
        return false
    end

    for _, player_id in ipairs(ordered_player_ids) do
        local player = players_by_id[player_id]
        if player.status == "Alive" and not player.is_ready then
            return false
        end
    end

    return true
end

function M.get_alive_count()
    return alive_count
end

function M.get_alive_player_ids()
    local player_ids = {}

    for _, player_id in ipairs(ordered_player_ids) do
        local player = players_by_id[player_id]
        if player.status == "Alive" then
            table.insert(player_ids, player_id)
        end
    end

    return player_ids
end

function M.spend_gold(player_id, amount, reason)
    require_non_negative_integer(amount, "amount")

    local player = players_by_id[player_id]
    if player == nil then
        error("Player not found: " .. tostring(player_id))
    end

    if amount == 0 then
        return
    end

    if player.gold < amount then
        error("Not enough gold for " .. tostring(reason))
    end

    player.gold = player.gold - amount

    table.insert(pending_events, {
        type = "GoldSpent",
        player_id = player_id,
        amount = amount,
        reason = reason,
        gold = player.gold,
    })
end

function M.grant_gold(player_id, amount, reason, wave)
    require_non_negative_integer(amount, "amount")

    local player = players_by_id[player_id]
    if player == nil then
        error("Player not found: " .. tostring(player_id))
    end

    if amount == 0 then
        return
    end

    player.gold = player.gold + amount

    local event = {
        type = "GoldGranted",
        player_id = player_id,
        amount = amount,
        reason = reason,
        gold = player.gold,
    }

    if wave ~= nil then
        event.wave = wave
    end

    table.insert(pending_events, event)
end

local function get_round_reward_amount(config, wave)
    local current_wave = wave or 1
    if type(current_wave) ~= "number"
        or current_wave < 1
        or current_wave % 1 ~= 0 then
        error("wave must be a positive integer: " .. tostring(wave))
    end

    local base_reward = config.round_reward
    local increment = config.round_reward_per_wave_increment or 0
    return base_reward + (current_wave - 1) * increment
end

function M.grant_round_reward(wave)
    for _, player_id in ipairs(ordered_player_ids) do
        local player = players_by_id[player_id]

        if player.status == "Alive" then
            local amount = get_round_reward_amount(
                active_player_configs[player_id],
                wave
            )
            M.grant_gold(player_id, amount, "RoundReward", wave)
        end
    end
end


local function copy_player(player)
    return {
        player_id = player.player_id,
        health = player.health,
        max_health = player.max_health,
        gold = player.gold,
        status = player.status,
        is_ready = player.is_ready,
    }
end

function M.get_snapshot()
    local players = {}
    for _, id in ipairs(ordered_player_ids) do
        local player = players_by_id[id]
        table.insert(players, copy_player(player))
    end

    return {
        alive_count =  alive_count,
        players = players,
    }
end

function M.drain_events()
    local events = pending_events
    pending_events = {}
    return events
end

function M.shutdown()
    reset()
end

return M
