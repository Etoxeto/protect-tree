local flow_config = require("Config.MatchFlow")

local M = {}

local phase_durations = {
    Preparation = flow_config.preparation_seconds,
    BossPreparation = flow_config.boss_preparation_seconds,
    Settlement = flow_config.settlement_seconds,
}

local current_phase = nil
local current_wave = 0
local remaining_seconds = 0
local is_started = false
local is_finished = false
local result = "InProgress"

local pending_events = {}

local function require_positive_integer(value, field_name)
    if type(value) ~= "number" or value <= 0 or value % 1 ~= 0 then
        error(field_name .. " must be a positive integer: " .. tostring(value))
    end
end

local function validate_config()
    require_positive_integer(flow_config.normal_wave_count, "normal_wave_count")
    require_positive_integer(flow_config.boss_wave, "boss_wave")

    if flow_config.boss_wave ~= flow_config.normal_wave_count + 1 then
        error("boss_wave must immediately follow normal waves")
    end

    for phase, duration in pairs(phase_durations) do
        if type(duration) ~= "number" or duration <= 0 then
            error("Phase duration must be positive: " .. phase)
        end
    end
end

local function get_phase_duration(phase)
    return phase_durations[phase] or 0
end

local function emit_phase_changed()
    local event = {
        type = "PhaseChanged",
        wave = current_wave,
        phase = current_phase,
    }

    if result ~= "InProgress" then
        event.result = result
    end

    table.insert(pending_events, event)
end

local function enter_phase(phase, wave)
    current_phase = phase
    current_wave = wave or current_wave
    remaining_seconds = get_phase_duration(phase)
    print("wave=" .. current_wave .. " phase=" .. current_phase)
    emit_phase_changed()
end

local function enter_next_timed_phase()
    if current_phase == "Preparation" then
        enter_phase("Battle")
        return
    end

    if current_phase == "Settlement" then
        if current_wave < flow_config.normal_wave_count then
            enter_phase("Preparation", current_wave + 1)
        else
            enter_phase("BossPreparation", flow_config.boss_wave)
        end
        return
    end

    if current_phase == "BossPreparation" then
        enter_phase("BossBattle")
        return
    end

    error("Timed phase cannot advance: " .. tostring(current_phase))
end

function M.start()
    validate_config()
    pending_events = {}
    is_started = true
    is_finished = false
    result = "InProgress"
    enter_phase("Preparation", 1)
end

function M.update(delta_time)
    if not is_started or is_finished then
        return
    end

    if current_phase == "Battle" or current_phase == "BossBattle" then
        return
    end

    remaining_seconds = remaining_seconds - delta_time

    while remaining_seconds <= 0 and not is_finished
        and current_phase ~= "Battle" and current_phase ~= "BossBattle" do
        local overflow = -remaining_seconds
        enter_next_timed_phase()
        remaining_seconds = remaining_seconds - overflow
    end
end

function M.complete_battle()
    if not is_started or is_finished or current_phase ~= "Battle" then
        return false
    end

    enter_phase("Settlement")
    return true
end

function M.complete_preparation()
    if not is_started or is_finished then
        return false
    end

    if current_phase ~= "Preparation" and current_phase ~= "BossPreparation" then
        return false
    end

    enter_next_timed_phase()
    return true
end

function M.end_match(match_result)
    if not is_started or is_finished then
        return false
    end

    if match_result ~= "Victory" and match_result ~= "Defeat" then
        error("Invalid match result: " .. tostring(match_result))
    end

    result = match_result
    is_finished = true
    enter_phase("End")
    return true
end

function M.complete_boss(match_result)
    if current_phase ~= "BossBattle" then
        return false
    end

    return M.end_match(match_result)
end

function M.shutdown()
    is_started = false
end

function M.get_snapshot()
    return {
        phase = current_phase,
        wave = current_wave,
        remaining_seconds = math.max(remaining_seconds, 0),
        is_finished = is_finished,
        result = result,
    }
end

function M.drain_events()
    local events = pending_events
    pending_events = {}
    return events
end

return M
